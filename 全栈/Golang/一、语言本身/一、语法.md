# 一、golang

### 变量

```go
package main

import "fmt"

// 1、go是静态语言
// 2、定义全局变量
// var name int = 1
var name = 1 // better

func main() {

	// 3、局部变量声明，局部变量被定义后必须使用，否则会报错
	age := 1
	fmt.Println(age)

	// 4、多变量定义，这个string可以不写
	var user1, user2, user3 string = "bobby1", "bobby2", "bobby3"
	fmt.Println(user1, user2, user3)

	/**
	1、要求变量的类型和赋值类型一致
	2、变量必须先定义才能使用
	3、变量名不能冲突
	4、简洁变量定义不能用于全局变量
	5、变量有默认值（零值）
	*/
}
```

### 常量

```go
func main() {
	const PI float32 = 3.1415926

	const (
		UNKNOWN = 1
		FALSE   = 2
		MALE    = 3
	)

	const (
		x int = 16
		y
		s = "abc"
		z
		m
	)

	fmt.Println(x, y, s, z, m) // 16 16 abc abc abc
}
```

### iota

```go
func main() {
	const (
		ERR1 = iota + 1
		ERR2
		ERR25 = "ha"
		ERR3
		ERR4 = iota
		ERR5 = iota
	)
	// 1 2 ha ha 5 6
	fmt.Println(ERR1, ERR2, ERR25, ERR3, ERR4, ERR5)

	// 遇到const，iota会归0
	const (
		WARN = iota
	)
	fmt.Println(WARN) // 0
}
```

### 匿名变量

```go
func a() (int, bool) {
	return 0, true
}

func main() {
	var _ int
	_, ok := a()
	if ok {
		fmt.Println(ok)
	}
}
```

### 数据类型

```go
package main

import "fmt"

func main() {
	var i8 int8
	var i16 int16
	var i32 int32
	var i64 int64
	var u8 uint8
	var u16 uint16
	var u32 uint32
	var u64 uint64
	var i int // 动态类型

	var f1 float32
	var f2 float32

	var b byte // 主要适用于存放字符，存放字符串
	var r rune // 也是字符，存中文

	var str string
}
```
