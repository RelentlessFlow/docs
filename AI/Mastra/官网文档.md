### 流式输出

```typescript
import { config } from 'dotenv';
import { z } from "zod";
import { chefAgent } from "./agents/chef";

config({ path: './.env' });

async function test() {
  const query =
    "我想做黄焖鸡，你能为我生成一个黄焖鸡的食谱吗？";
  console.log(`查询: ${query}`);
 
  const stream = await chefAgent.stream([{ role: "user", content: query }]);

  console.log("\n Chef Michel: ");

  for await (const chunk of stream.textStream) {
    process.stdout.write(chunk);
  }

  console.log("\n\n✅ Recipe complete!");
}


test();
```

## 股票Agent案例

**创建一个 Tools**

```typescript
import { createTool } from "@mastra/core/tools";
import { z } from "zod";
 
const getStockPrice = async (symbol: string) => {
  const data = await fetch(
    `https://mastra-stock-data.vercel.app/api/stock-data?symbol=${symbol}`,
  ).then((r) => r.json());
  return data.prices["4. close"];
};

export const stockPrices = createTool({
  id: "获取股票价格",
  inputSchema: z.object({
    symbol: z.string(),
  }),
  description: `获取给定股票代码的前一天收盘价`,
  execute: async ({ context: { symbol } }) => {
    console.log("使用工具获取股票价格，股票代码为", symbol);
    return {
      symbol,
      currentPrice: await getStockPrice(symbol),
    };
  },
});
```

1。inputSchema 中定义的数据，在execute的context中读取

2、execute 返回的数据，可以添加到大模型上下文中，这里可以读取其他的函数

**创建一个Agent**

```typescript
import { Agent } from "@mastra/core/agent";
import { openai } from "@ai-sdk/openai";
import { stockPrices } from "../tools/stockPrices";
  
export const stockAgent = new Agent({
  name: "股票助手",
  instructions:
    "你是一个有帮助的助手，提供当前股票价格。当被问到某只股票时，使用股票价格工具来获取该股票的价格。",
  model: openai("gpt-4o-mini"),
  tools: {
    stockPrices,
  },
});
```

**创建一个测试类**

```typescript
import { config } from 'dotenv';
import { stockAgent } from "./agents/stock";

config({ path: './.env' });

async function test() {
  const query =
    "Apple股票目前的价格（AAPL）？";
  console.log(`查询: ${query}`);
 
  const response = await stockAgent.generate(
    [{ role: "user", content: query }],
  );

  console.log(response.text);
}

test();
```

这里测试类，不知为何，tools里面的execute会走两遍，但是如果以接口的形式调用就不会。

## Memory

mastra 通过 thread_id 组织对话，通过 resource_id 表示与之对话的用户

```json
// PATH
/api/agents/{agentId}/generate

// Body
{
    "messages": [
        {
            "role": "user",
            "content": "我需要学习下编程，JavaScript是什么？它怎么编写HelloWorld？"
        }
    ],
    "threadId": "user_1_chat_1",
    "resourceId": "user_1"
}
```

使用这个功能，需要为Agent配置下memory

```typescript
const memory = new Memory() // 默认使用嵌入式数据库 LibSQL（SQLite的一种）
```

Memory 默认会保存近 40条数据，使用lastMessages字段可以进行配置。

它内存采用了向量数据库mastra进行予以检索，这里可以自定义嵌入模型、向量数据库等。

Memory 可以使用自定义数据库，这样便于对历史对话、用户数据进行管理

```typescript
import { Memory } from "@mastra/memory";
import { PostgresStore, PgVector } from "@mastra/pg";
 
const memory = new Memory({
  storage: new PostgresStore({
    host: "localhost",
    port: 5432,
    user: "postgres",
    database: "postgres",
    password: "postgres",
  }),
  vector: new PgVector("postgresql://user:pass@localhost:5432/db"),
  options: {
    // Number of recent messages to include (false to disable)
    lastMessages: 10,
    // Configure vector-based semantic search
    semanticRecall: {
      topK: 3, // Number of semantic search results
      messageRange: 2, // Messages before and after each result
    },
  },
});
```

这里使用DataGrip查询它的数据，它生成了五张表

mastra_traces

## 推理模型

```
import { codeAgent } from "./agents";
import { config } from 'dotenv'

config({ path: '.env' })

const stream = await codeAgent.stream('你是谁？', {
  onStepFinish: (step) => {
    console.log('[推理步骤]', step);
  },
});


for await (const chunk of stream.fullStream) {
  if (chunk.type === 'reasoning') {
    process.stdout.write(chunk.textDelta);
  }
  if (chunk.type === 'text-delta') {
    process.stdout.write(chunk.textDelta);
  }
}
```
