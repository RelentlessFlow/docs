TS 生成任意大小的文件

```typescript
import { createWriteStream } from 'fs';

class FileGenerator {
    constructor(private sizeInMB: number, private outputPath: string) {}

    async generate(): Promise<void> {
        const writeStream = createWriteStream(this.outputPath);
        const chunkSize = 1024 * 1024; // 1MB in bytes
        const buffer = Buffer.alloc(chunkSize, 'a fixed string to fill the buffer\n');

        for (let i = 0; i < this.sizeInMB; i++) {
            await this.writeChunk(writeStream, buffer);
        }

        writeStream.end();
        console.log(`File of ${this.sizeInMB}MB has been created at ${this.outputPath}`);
    }

    private writeChunk(writeStream: any, buffer: Buffer): Promise<void> {
        return new Promise((resolve, reject) => {
            writeStream.write(buffer, (error: Error | null) => {
                if (error) {
                    reject(error);
                } else {
                    resolve();
                }
            });
        });
    }
}

// 使用示例：生成一个5MB大小的文件
const generator = new FileGenerator(4, './outputFile.txt');
generator.generate().then(() => console.log('Finished generating file.'))
.catch((error) => console.error('Error occurred:', error));
```
