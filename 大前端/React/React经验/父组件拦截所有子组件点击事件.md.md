```react
export default class PreventClick extends Component {
  constructor(props) {
    super(props);
    this.myRef = createRef(null);
  }
  componentDidMount() {
    this.myRef.current.addEventListener('click', (e) => {
      e.stopPropagation()
    });
  }
  render() {
    return (
      <div ref={this.myRef}>{this.props.children}</div> )
  }
}
```
